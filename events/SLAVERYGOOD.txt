namespace = SLAVERYGOOD

country_event = { #Matter of the Naghali
	id = SLAVERYGOOD.0
	title = SLAVERYGOOD.0t
	desc = SLAVERYGOOD.0d
	is_triggered_only = yes
	
	option = { #We shall expand the practice of mukataba, or manumission.
		name = SLAVERYGOOD.0a
		trigger = {
			has_country_flag = naghali_repression
		}
		clr_country_flag = considering_naghali_laws
		clr_country_flag = naghali_repression
		set_country_flag = naghali_enfranchisement
		every_owned_province = {
			limit = { culture = naghali }
			remove_province_modifier = naghali_discontent
		}
	}
	
	option = { #The Naghali and their people are inherently our inferiors!
		name = SLAVERYGOOD.0b
		trigger = {
			has_country_flag = naghali_enfranchisement
		}
		clr_country_flag = considering_naghali_laws
		clr_country_flag = naghali_enfranchisement
		set_country_flag = naghali_repression
		every_owned_province = {
			limit = { culture = naghali }
			add_permanent_province_modifier = {
				name = naghali_discontent
				duration = -1
			}
		}
	}
	
	option = { #This all sounds very complex...
		name = SLAVERYGOOD.0c
		trigger = {
			OR = {
				has_country_flag = naghali_enfranchisement
				has_country_flag = naghali_repression
			}
		}
		clr_country_flag = considering_naghali_laws
	}
}

province_event = { #Rise of a Naghali Population Hub
	id = SLAVERYGOOD.1
	title = SLAVERYGOOD.1t
	desc = SLAVERYGOOD.1d
	trigger = {
		has_trade_building_trigger = yes
		owner = {
			has_country_flag = naghali_enfranchisement
			NOT = {
				any_owned_province = { has_province_modifier = naghali_confluence }
				any_owned_province = { has_province_modifier = naghali_concessions }
			}
		}
	}
	mean_time_to_happen = {
		years = 25
	}
	
	option = { #Enrich yourselves, and you shall prosper.
		name = SLAVERYGOOD.1a
		change_culture = naghali
		add_core = GHL
		add_province_modifier = {
			name = naghali_confluence
			duration = 3650
		}
	}
	
	option = { #What luck, we have room for new settlement just over there!
		name = SLAVERYGOOD.1b
		trigger = {
			owner = {
				any_owned_province = {
					culture = naghali
					NOT = { has_province_modifier = naghali_confluence }
				}
			}
		}
		owner = {
			random_owned_province = {
				limit = {
					culture = naghali
					NOT = { has_province_modifier = naghali_confluence }
				}
				add_base_tax = 1
				add_base_production = 1
				add_base_manpower = 1
				add_province_modifier = {
					name = naghali_confluence
					duration = 3650
				}
			}
		}
	}
	
	option = { #Allow the [Root.Owner.Monarch.GetTitle] to help you get started.
		name = SLAVERYGOOD.1c
		trigger = { owner = { treasury = 100 } }
		owner = { add_treasury = -100 }
		change_culture = naghali
		add_province_modifier = {
			name = naghali_confluence
			duration = 3650
		}
		add_base_tax = 1
		add_base_production = 1
		add_base_manpower = 1
	}
}

}
