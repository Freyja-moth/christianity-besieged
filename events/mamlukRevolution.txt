namespace = mamlukRevolt
country_event = { #Mamluks took over the government
    id = mamlukRevolt.0
    title = mamlukRevolt.t
    desc = mamlukRevolt.d
    is_triggered_only = yes
    immediate = {
        hidden_effect = {
            BTA = {
                country_event = {
                    id = mamlukRevolt.1
                }
            }
        }
    }
    option = {
        name = mamlukRevolt.a
        hidden_effect = {
            if = {
                limit = {
                    NOT = { government = monarchy }
                }
                change_government = monarchy
            }
            set_saved_name = {
                key = mamluk_sultan
                type = advisor
                scope = ROOT
            }
            define_ruler = {
                name = mamluk_sultan
                culture = ROOT
                mil = 4
                age = 30
            }    
            lose_reforms = 2
        }
        override_country_name = THE_MAMLUKS
        add_government_reform = mamluk_government
        grant_independence = yes
        remove_country_modifier = mamluks_revolted
    }
}

country_event = { #Decides target for Mamluk Revolt HIDDEN
    id = mamlukRevolt.2
    title = mamlukRevolt.2t
    desc = mamlukRevolt.2d
    hidden = yes
    trigger = {
        reform_desire = 0.95
        religion = ashabi
        NOT = { has_reform = mamluk_government }
        NOT = { tag = event_target:mamlukRevolting }
    }
    mean_time_to_happen = {
        months = 120

        modifier = {
            factor = 0.5
            NOT = { stability = -1 }
        }
        modifier = {
            factor = 50
            is_at_war = yes
        }
        modifier = {
            factor = 0.5
            government = monarchy
            NOT = { legitimacy = 30 }
        }
        modifier = {
            factor = 1000
            government = theocracy
        }
        modifier = {
            factor = 0.5
            government = republic
            NOT = { legitimacy_equivalent = 30 }
        }
        modifier = {
            factor = 0.1
            any_hired_mercenary_company = {
                template = mamlukean_guard
            }
        }
    }
    option = {
        name = mamlukRevolt.2a
        capital_scope = {
            spawn_rebels = {
                type = mamluk_revolt
                size = 2
            }
        }
        save_global_event_target_as = mamlukRevolting
        add_reform_desire = -0.3
        country_event = { id = mamlukRevolt.3 }
    }
}
