# default options are 0-based
# In console, can use imperial_incident [key] to immediately begin an incident

incident_shadow_kingdom = {
	event = shadow_kingdom.2
	default_option = 0
	
	can_stop = {
		always = no
	}
	
	0 = { # give up
		factor = 2
		modifier = {
			factor = 5
			capital_scope = {
				region = france_region
			}
		}
		modifier = {
			factor = 0
			AND = {
				alliance_with = emperor
				NOT = {
					capital_scope = {
						region = france_region
					}
					development_in_provinces = {
						value = 50
					}
				}
			}
		}
	}
	
	1 = { # rein in
		factor = 1
		modifier = {
			factor = 4
			has_opinion = {
				who = emperor
				value = 50
			}
		}
		modifier = {
			factor = 0
			OR = {
				NOT = {
					religion = german_pagan
				}
				NOT = {
					culture_group = germanic_group
				}
			}
		}
	}
}

incident_pu_subject_joins_empire = {
	event = incident_generic.2
	default_option = 0
	
	can_stop = {
		OR = {
			NOT = {
				any_country = {
					has_country_flag = current_pu_incident_target_flag
				}
			}
			any_country = {
				has_country_flag = current_pu_incident_target_flag
				OR = {
					is_subject = no
					overlord = {
						is_part_of_hre = no
					}
				}
			}
		}
	}
	
	0 = { # join
		factor = 1
		modifier = {
			factor = 1000
			any_subject_country = {
				has_country_flag = current_pu_incident_target_flag
			}
		}
	}
	
	1 = { # denied
		factor = 1
	}
}

incident_great_peasants_war = {
	event = great_peasant_war.3
	default_option = 0
	
	can_stop = {
		always = no
	}
	
	0 = { # crush peasants
		factor = 1
		modifier = {
			factor = 10
			estate_influence = {
				estate = estate_nobles
				influence = 60
			}
		}
	}
	
	1 = { # concessions
		factor = 1
		modifier = {
			factor = 1000
			has_reform = peasants_republic
		}
		modifier = {
			factor = 10
			NOT = {
				estate_influence = {
					estate = estate_nobles
					influence = 40
				}
			}
		}
	}
}
