irish_high_kingdom_slot_3 = {
	slot = 3
	generic = no
	ai = yes
	has_country_shield = no

	potential = {
		has_reform = irish_high_kingdom
	}

	weather_pretenders_mission = {
		icon = mission_rb_conquer_the_highlands

		required_missions = {
			proclaim_high_king_of_ireland_mission
		}		

		trigger = {
			NOT = {
				any_country = {
					if = { 
						limit = { 
							NOT  = { tag = ROOT } 
						}
					}
					OR = {
						has_casus_belli = {
							type = illegitimate_high_king_of_ireland_cb
							target = ROOT
						}
						has_reform = irish_high_kingdom
					}
				}
			}
		}

		effect = {
			if = {
				limit = { has_country_flag = ousted_pretender_high_king }
				every_country = {
					limit = { has_country_flag = pretender_high_king }

					vassalize = ROOT
				}
				add_prestige = 20
			}

			if = {
				limit = { has_country_flag = defended_high_king_legitimacy }
				every_country = {
					limit = { has_country_flag = forced_to_recognise_high_king }

					vassalize = ROOT
				}
				add_prestige = 10
			}

			every_country = {
				limit = { has_reform = irish_kingdom }

				add_casus_belli = {
					type = subjugate_lesser_irish_kingdom_cb
					target = THIS
				}
			}
		}
	}
}
